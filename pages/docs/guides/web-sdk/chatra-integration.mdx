

Дополните кофнигурацию Chatra

```html
<!-- Chatra {literal} -->
<script>
window.ChatraSetup = {
  colors: {
    buttonText: '#ffffff',
    buttonBg: '#8c00ff'
  },
  onAnalyticEvent: function(eventName) {
    (function(){
      if (window.rstat){
        name = eventName.lowerCase().replace(/\s+/g, '_')
        window.rstat('onReady', function(){
            window.rstat.request('chatra', name)
        })
      }
    })();     
  }
};
(function(d, w, c) {
    w.ChatraID = 'Jc8cBDRCMZR5RAHo6';
    var s = d.createElement('script');
    w[c] = w[c] || function() {
        (w[c].q = w[c].q || []).push(arguments);
    };
    s.async = true;
    s.src = 'https://call.chatra.io/chatra.js';
    if (d.head) d.head.appendChild(s);
})(document, window, 'Chatra');
</script>
<!-- /Chatra {/literal} -->
```
